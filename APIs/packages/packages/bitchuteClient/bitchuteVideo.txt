#{
	Returns the raw video file from a BitChute video ID.
	
	Usage example:
		$rawVideoFile = bitchuteVideoFileUrl: "someVideoid".
	
	This URL could be used for downloading or playing the video.
	If you want to do the second, check out "uiBitchutePlayer".
}
$bitchuteVideoFileUrl = ($id) fun {
	second: ({src\s*=\s*"([^"]+} + id + {[^"]+)}) match loadGlobal: bitchuteEmbedUrl + id
} ? Video
