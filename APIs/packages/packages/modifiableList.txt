$modifiableList = ($_normalList ? (List:Any)) fun {
	!ifElse ((first each _normalList) eq $push;$getArray;$clear;$setOnAddition;$setOnRemoval;$setOnChange;$addToEnd;$removeEquals;$removeIndex;$removeMatch) {
		_normalList
	};{
		$listVar ? (List:Any) = _normalList. $listVarRef = reference: $listVar.
		$onAddition = (). $onAdditionRef = reference: $onAddition.
		$onRemoval = (). $onRemovalRef = reference: $onRemoval.
		$onChange = (). $onChangeRef = reference: $onChange.
		$listEnd = (). $listEndRef = reference: $listEnd.
		
		@($uiGenerator ; uiLiveListView)
		["push"; ($_items) fun {
			listVarRef = listVar ; _items.
			onAddition:_items.
			!onChange.
			listVar
		}];
		["getArray"; () fun {listVar;listEnd}];
		["clear";() fun {listVarRef = ()}];
		["setOnAddition";($function) fun {onAdditionRef = function}];
		["setOnRemoval";($function) fun {onRemovalRef = function}];
		["setOnChange";($function) fun {onChangeRef = function}];
		["addToEnd";($_endItems) fun {listEndRef = listEnd;_endItems}];
		["removeEquals";($_item) fun {
			listVarRef = {not: _item eq x} filter listVar.
			!onRemoval.
			!onChange.
			listVar
		}];
		["removeIndex";($index) fun {
			listVarRef = ({not: index eq x} filter arrayIndexes) at listVar.
			!onRemoval.
			!onChange.
			listVar
		}];
		["removeMatch";($function) fun {
			listVarRef = {not: function: x} filter listVar.
			!onRemoval.
			!onChange.
			listVar
		}]
	}
}.
