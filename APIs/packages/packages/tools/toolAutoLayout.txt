$toolsAtLeft = !fun ($_subTools) {
	`Desktop`
	registerDesktopTool: {
		@(percentWidth 75) @left uiContainer: 5 uiAddHorizontalPxSpaces {do:$desktop propOf x} each _subTools
	}.
	
	`Mobile`
	($_subTool fun {
		($mobile propOf _subTool) registerMobileTool ($title propOf _subTool)
	}) each _subTools.
}.
$toolAtLeft = !fun ($_tool) {
	`Desktop`
	registerDesktopTool: {
		@left do:$desktop propOf _tool
	}.
	
	`Mobile`
	($mobile propOf _tool) registerMobileTool ($title propOf _tool).
}.
$toolAtRight = !fun ($_tool) {
	`Desktop`
	registerDesktopTool: {
		@right do:$desktop propOf _tool
	}.
	
	`Mobile`
	($mobile propOf _tool) registerMobileTool ($title propOf _tool).
}.
