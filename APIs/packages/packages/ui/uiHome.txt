$uiHomeScreenIcon = ($iconGraphic ? UiElement ; $_applySpecificLook ? (default: {@(backgroundColor red) x})) fun {
	$ico = @rounded @(emWidth 5) @(emHeight 5) iconGraphic.
	_applySpecificLook: ico
}.
$uiSimpleHomeScreenApp = ($description ? String ; $action) fun {
	$color = randomItem: [red];[green];[blue].
	$gradient = backgroundGradientTopToBottom: [color];[{max: 0;x} each color - 150].
	$border = borderColor: {max: 0;x} each color - 50.
	@(onTap action) @centerText uiContainer: ((uiContainer: @(emFontSize 1,5) uiText: first of "" strSplit description) uiHomeScreenIcon {@(foregroundColor white) @border @gradient x});uiText: description
}.
$uiHomeScreen = ($_action ? (default: launchAppName)) fun {
	`Load the app data from JSON`
	$apps = (parseJson: loadLocal: "/apps/home/config/standardApps.json");
		!if developerMode {
			parseJson: loadLocal: "/apps/home/config/developerApps.json"
		}.
	
	uiColumns: ($_app fun {@(percentWidth 100) (second of _app) uiSimpleHomeScreenApp () fun {_action: first of _app}}) each apps.
}.
